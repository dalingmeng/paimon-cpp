62 template <typename T>
63 static Result<T> VisitPredicate(const std::shared_ptr<LeafPredicate>& predicate,
64 const std::shared_ptr<FunctionVisitor<T>>& visitor) {
65 auto type = predicate->GetFunction().GetType();
66 switch (type) {
67 case Function::Type::IS_NULL:
68 return visitor->VisitIsNull();
69 case Function::Type::IS_NOT_NULL:
70 return visitor->VisitIsNotNull();
71 case Function::Type::EQUAL: {
72 assert(predicate->Literals().size() == 1);
73 return visitor->VisitEqual(predicate->Literals()[0]);
74 }
75 case Function::Type::NOT_EQUAL: {
76 assert(predicate->Literals().size() == 1);
77 return visitor->VisitNotEqual(predicate->Literals()[0]);
78 }
79 case Function::Type::GREATER_THAN: {
80 assert(predicate->Literals().size() == 1);
81 return visitor->VisitGreaterThan(predicate->Literals()[0]);
82 }
83 case Function::Type::GREATER_OR_EQUAL: {
84 assert(predicate->Literals().size() == 1);
85 return visitor->VisitGreaterOrEqual(predicate->Literals()[0]);
86 }
87 case Function::Type::LESS_THAN: {
88 assert(predicate->Literals().size() == 1);
89 return visitor->VisitLessThan(predicate->Literals()[0]);
90 }
91 case Function::Type::LESS_OR_EQUAL: {
92 assert(predicate->Literals().size() == 1);
93 return visitor->VisitLessOrEqual(predicate->Literals()[0]);
94 }
95 case Function::Type::IN:
96 return visitor->VisitIn(predicate->Literals());
97 case Function::Type::NOT_IN:
98 return visitor->VisitNotIn(predicate->Literals());
99 case Function::Type::STARTS_WITH: {
100 assert(predicate->Literals().size() == 1);
101 return visitor->VisitStartsWith(predicate->Literals()[0]);
102 }
103 case Function::Type::ENDS_WITH: {
104 assert(predicate->Literals().size() == 1);
105 return visitor->VisitEndsWith(predicate->Literals()[0]);
106 }
107 case Function::Type::CONTAINS: {
108 assert(predicate->Literals().size() == 1);
109 return visitor->VisitContains(predicate->Literals()[0]);
110 }
111 case Function::Type::LIKE: {
112 assert(predicate->Literals().size() == 1);
113 return visitor->VisitLike(predicate->Literals()[0]);
114 }
115 default:
116 return Status::Invalid("invalid " + predicate->GetFunction().ToString() +
117 " function in leaf predicate");
118 }
119 }
